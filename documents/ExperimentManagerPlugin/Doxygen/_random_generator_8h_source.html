57  int IRandom(int min, int max)
58  {// output random integer
59  if(min == max)
60  return min;
61  // get integer random number in desired interval
62  int iinterval = max - min;// + 1;
63  if (iinterval < 0)//<= 0)
64  return 0x80000000; // error
65  int r = int((iinterval+1) * Random()); // truncate, Random() returns <1.0
66  //if (r >= iinterval)
67  // return max;//r = iinterval-1;
68  return min + r;
69  }
70 
71  double DRandom(double min, double max)
72  {// output random double
73  if(min == max)
74  return min;
75  // get double random number in desired interval
76  double iinterval = max - min;// + 1;
77  if (iinterval <= 0)//<= 0)
78  return 0x80000000; // error
79  //double dsmall = DBL_MIN;
80  double r = (iinterval+DBL_MIN) * Random(); // truncate
81  //if (r >= iinterval)
82  // return max;// r = iinterval-1;
83  return min + r;
84  }
