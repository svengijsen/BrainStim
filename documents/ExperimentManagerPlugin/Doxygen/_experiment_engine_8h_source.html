134  template< typename T1 > QVariant stringToTemplateVariabele(const QString &sVar)
135  {
136  bool bResult = false;
137  if (typeid(T1) == typeid(QString))
138  {
139  return sVar;
140  }
141  else if (typeid(T1) == typeid(QStringList))
142  {
143  return sVar.split(",");
144  }
145  else if (typeid(T1) == typeid(QColor))
146  {
147  return QColor(sVar);
148  }
149  else if (typeid(T1) == typeid(int))
150  {
151  int tmpInt = sVar.toInt(&bResult);
152  if (bResult)
153  return tmpInt;
154  }
155  else if (typeid(T1) == typeid(float))
156  {
157  float tmpFloat = sVar.toFloat(&bResult);
158  if (bResult)
159  return tmpFloat;
160  }
161  else if (typeid(T1) == typeid(double))
162  {
163  double tmpDouble = sVar.toDouble(&bResult);
164  if (bResult)
165  return tmpDouble;
166  }
167  else if (typeid(T1) == typeid(bool))
168  {
169  if (TYPE_BOOL_TRUE)
170  {return true;}
171  else if (TYPE_BOOL_FALSE)
172  {return true;}
173  }
174  qDebug() << __FUNCTION__ << "Could not parse the template variable(" << sVar << ")";
175  return T1();
176  }
