845  else if(sValue.toLower() == METHOD_TYPE_SLOT_TAG)

847  else
848  return ExperimentStructuresNameSpace::MethodType::METHOD_TYPE_UNDEFINED;
849  };

857  {
858  if(nValue == ExperimentStructuresNameSpace::MethodType::METHOD_TYPE_SIGNAL)

861  return METHOD_TYPE_SLOT_TAG;
