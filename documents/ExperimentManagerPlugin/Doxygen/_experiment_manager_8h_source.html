167  template< typename T > T* getExperimentObjectVariabelePointer(const int &nObjectID,const QString &sKeyName)
168  {
169  if (mExperimentObjectList.contains(nObjectID))
170  {
171  if ((mExperimentObjectList.value(nObjectID).ExpBlockParams == NULL) || (mExperimentObjectList.value(nObjectID).ExpBlockParams->isEmpty()))
172  return NULL;
173  if (mExperimentObjectList.value(nObjectID).ExpBlockParams->contains(sKeyName.toLower()))
174  {
175  if (mExperimentObjectList.value(nObjectID).typedExpParamCntnr)
176  return mExperimentObjectList.value(nObjectID).typedExpParamCntnr->getPropertySetting<T>(sKeyName);
177  else
178  return NULL;
179  }
180  }
181  return NULL;
182  }

184  template< typename T > bool insertExperimentObjectVariabelePointer(const int &nObjectID,const QString &sKeyName,T &tVariabele, bool bCreateVariabeleInMemory = false)
185  {
186  if (mExperimentObjectList.contains(nObjectID))
187  {
188  objectElement *mTmpObjectElement;
189  mTmpObjectElement = &(mExperimentObjectList[nObjectID]);
190  if ((mTmpObjectElement->ExpBlockParams == NULL) || (mTmpObjectElement->ExpBlockParams->isEmpty()))
191  return false;
192  if (mTmpObjectElement->typedExpParamCntnr == NULL)
193  mTmpObjectElement->typedExpParamCntnr = new TypedPropertySettingContainer();
194  bool bRetVal;
195  if(bCreateVariabeleInMemory)
196  {
197  bRetVal = mTmpObjectElement->typedExpParamCntnr->createPropertySetting<T>(sKeyName, tVariabele);
198  }
199  else
200  {
201  bRetVal = mTmpObjectElement->typedExpParamCntnr->insertPropertySetting<T>(sKeyName, &tVariabele);
202  }
203  return bRetVal;
204  }
205  return false;
206  }
